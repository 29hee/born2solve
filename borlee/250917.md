#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

int* solution(int arr[], size_t arr_len) 
{
    // return 값은 malloc 등 동적 할당을 사용해주세요. 할당 길이는 상황에 맞게 변경해주세요.
    int* stk = (int*)malloc(sizeof(int) * arr_len);
    if (!stk)
        return NULL;
    size_t top = 0;
    for (size_t i = 0; i < arr_len;)
    {
        if (top == 0)
        {
            stk[top++] = arr[i];
            i++;
        }
        else if (stk[top - 1] < arr[i])
        {
            stk[top++] = arr[i];
            i++;
        }
        else
            top--;
    }
        
    return stk;
}
