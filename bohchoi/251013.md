#include <string>
#include <vector>
#include <algorithm>

using namespace std;

vector<string> solution(string myString) {
    vector<string> answer;
    string str = "";
    for (int i = 0; i < myString.size(); i++)
    {
        if (!str.empty() && myString.at(i) == 'x')
        {
            answer.push_back(str);
            str.clear();
        }
        else if (myString.at(i) != 'x')
            str += myString.at(i);
    }
    if (!str.empty()) 
        answer.push_back(str);
    sort(answer.begin(), answer.end());
    
    return answer;
}
