#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> arr) {
    vector<int> done(1, -1);
    vector<int> last_two;
    bool    flag = false;
    int     i;
    
    for (i = 0; i < arr.size(); i++)
        if (arr.at(i) == 2)
        {
            last_two.push_back(i);
            flag = true;
        }
    if (!flag)
        return done;
    
    done.clear();    
    for (i = last_two.front(); i < last_two.back() + 1; i++)
        done.push_back(arr.at(i));
    return done;
}
