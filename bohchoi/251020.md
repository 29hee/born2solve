#include <string>
#include <vector>

using namespace std;

string solution(string n_str) {
    string answer = "";
    int i = 0;
    int flag = 0;
    
    while (i < n_str.size())
    {
        if (!flag && n_str.at(i) == '0')
        {
            i++;
        }
        else
        {
            answer += n_str.at(i);
            i++;
            flag = 1;
        }
    }
    return answer;
}
