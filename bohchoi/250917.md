#include <string>
#include <vector>
#include <cmath>
#include <algorithm>
using namespace std;

bool type_one(int a, int b, int c, int d)
{      
    if ((a == b && b == c) && a != d)
        return true;
    if ((a == b && b == d) && a != c)
        return true;
    if ((a == c && c == d) && a != b)
        return true;
    if ((b == c && c == d) && b != a)
        return true;
    return false;
}

int type_two(int a, int b, int c, int d)
{   
    if (a == b && c == d && a != c)
        return (a + c) * abs(a - c);
    if (a == c && b == d && a != b)
        return (a + b) * abs(a - b);
    if (a == d && b == c && a != b)
        return (a + b) * abs(a - b);
    return 0;
}

int type_three(int a, int b, int c, int d)
{
    if (a == b && c != d && a != c && a != d)
        return c * d;
    if (a == c && b != d && a != b && a != d)
        return b * d;
    if (a == d && b != c && a != b && a != c)
        return b * c;
    if (b == c && a != d && b != a && b != d)
        return a * d;
    if (b == d && a != c && b != a && b != c)
        return a * c;
    if (c == d && a != b && c != a && c != b)
        return a * b;
    return 0;
}

int type_four(int a, int b, int c, int d)
{
    return min({a, b, c, d});
}

int solution(int a, int b, int c, int d) {
    
    // 1. 네 주사위가 모두 같은 경우
    if (a == b && b == c && c == d)
        return (1111 * a);
    
    // 2. 세 개가 같고 하나가 다른 경우
    if (type_one(a, b, c, d)) {
        if (a == b && b == c)
            return pow((10 * a + d), 2);
        if (a == b && b == d)
            return pow((10 * a + c), 2);
        if (a == c && c == d)
            return pow((10 * a + b), 2);
        if (b == c && c == d)
            return pow((10 * b + a), 2);
    }
    
    // 3. 두 개씩 같은 경우
    if (type_two(a, b, c, d) != 0)
        return type_two(a, b, c, d);
    
    // 4. 한 쌍과 두 개의 서로 다른 숫자
    if (type_three(a, b, c, d) != 0)
        return type_three(a, b, c, d);
    
    // 5. 모두 다른 경우
    if (a != b && a != c && a != d && b != c && b != d && c != d)
        return type_four(a, b, c, d);
    
    return 0;
}
