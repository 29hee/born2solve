#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> arr, vector<bool> flag) {
    vector<int> answer;
    
    for (int i = 0; i < arr.size(); i++)
    {
        if (flag.at(i))
            for (int count = 0; count < arr.at(i); count++)
            {
                answer.push_back(arr.at(i));
                answer.push_back(arr.at(i));
            }
        else
            for (int count = 0; count < arr.at(i); count++)
                answer.pop_back();
    }
    
    return answer;
}
