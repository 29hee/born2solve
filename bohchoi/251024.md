**오늘의 메모(선택)**: 

**코드(선택)**:

```cpp

#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> num_list, int n) {
    vector<int> answer;
    
    for (int i = 0; i < n; i++)
    {
        answer.insert(answer.begin() + i, num_list[i]);
    }
    return answer;
}

```
