#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int solution(vector<string> strArr) {
    int Max = 1;
    int count = 1;
    vector<int> num_list;
    
    for (int i = 0; i < strArr.size(); i++)
        num_list.push_back(strArr[i].size());
    
    sort(num_list.begin(), num_list.end());
    
    int i = 1;
    while (i < num_list.size())
    {
        if (num_list.at(i - 1) == num_list.at(i))
            count++;
        else
        {
            if (Max < count)
                Max = count;
            count = 1;
        }
        i++;
    }
    if (Max < count)
        Max = count;
    return Max;
}
