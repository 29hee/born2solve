#include <algorithm>
#include <cmath>
#include <cstring>

#define TAEBKIM                                                                \
    std::ios::sync_with_stdio(false);                                          \
    std::cin.tie(NULL);                                                        \
    std::cout.tie(NULL);

;

using namespace std;

int number[8001];

void solve() {
    int n, num;
    long long sum = 0;

    int maximum = -4001;
    int minimum = 4001;
    int max_freq = 0;
    int median = 0;
    int mode = 0;

    cin >> n;

    for (int i = 0; i < n; i++) {
        cin >> num;
        sum += num;
        number[num + 4000]++;

        if (num > maximum)
            maximum = num;
        if (num < minimum)
            minimum = num;
    }

    double avg = (double)sum / n;
    int rounded_avg = (int)round(avg);

    if (rounded_avg == 0) {
        cout << 0 << '\n';
    } else {
        cout << rounded_avg << '\n';
    }

    int count_sum = 0;
    for (int i = 0; i < 8001; i++) {
        count_sum += number[i];
        if (count_sum >= n / 2 + 1) {
            median = i - 4000;
            break;
        }
    }
    cout << median << '\n';

    vector<int> mode_candidates;

    for (int i = 0; i < 8001; i++) {
        if (number[i] > max_freq) {
            max_freq = number[i];
        }
    }

    for (int i = 0; i < 8001; i++) {
        if (number[i] == max_freq) {
            mode_candidates.push_back(i - 4000);
        }
    }

    if (mode_candidates.size() == 1) {
        mode = mode_candidates[0];
    } else {
        mode = mode_candidates[1];
    }
    cout << mode << '\n';

    cout << maximum - minimum << '\n';
}

int main() {
    TAEBKIM;

    solve();

    return 0;
}
