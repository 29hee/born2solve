#define TAEBKIM                                                                \
    std::ios::sync_with_stdio(false);                                          \
    std::cin.tie(NULL);                                                        \
    std::cout.tie(NULL);

long long gcd(int a, int b) {
    if (a % b == 0)
        return b;
    return gcd(b, a % b);
}

long long solve(int *arr, int n) {
    long long sum = 0;

    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            sum += gcd(arr[i], arr[j]);
        }
    }

    return sum;
}

int main() {
    TAEBKIM;

    int t;
    std::cin >> t;

    int n;
    int arr[101];

    while (t--) {
        std::cin >> n;
        for (int i = 0; i < n; i++) {
            std::cin >> arr[i];
        }

        std::cout << solve(arr, n) << "\n";
    }

    return 0;
}
