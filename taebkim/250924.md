#include <algorithm>
#include <cmath>
#include <iomanip>

#include <stack>

#include <string>
#include <unordered_set>
#include <utility>
#include <vector>

#define TAEBKIM                                                                \
    std::ios::sync_with_stdio(false);                                          \
    std::cin.tie(NULL);                                                        \
    std::cout.tie(NULL);

std::priority_queue<int, std::vector<int>, std::less<int>> q;

void print_queue() {
    if (q.empty()) {
        std::cout << "0" << "\n";
    } else {
        std::cout << q.top() << "\n";
        q.pop();
    }
}

int main() {
    TAEBKIM;

    int n;
    std::cin >> n;
    int x;
    for (int i = 0; i < n; i++) {
        std::cin >> x;
        if (x == 0) {
            print_queue();
        } else {
            q.push(x);
        }
    }

    return 0;
}
