**오늘의 메모(선택)**: 

백준 레벨업을 위한 쉬운단계 푸는중입니다. 근데 어렵네요.


**코드(선택)**:

```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int v[31] = {0};

    for (int i = 0; i < 28; i++) {
        int num;
        cin >> num;
        v[num] = 1;
    }

    int target1 = 0, target2 = 0;
    for (int i = 1; i <= 30; i++) {
        if (v[i] == 0) {
            if (target1 == 0) target1 = i;
            else target2 = i;
        }
    }

    cout << target1 << "\n" << target2 << "\n";
    return 0;
}
