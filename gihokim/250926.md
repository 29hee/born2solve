**코드(선택)**:

```c++
#include <set>

class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        set<int> s;

        for (int num : nums) {
            s.insert(num);
        }

        nums.clear();

        for (int num : s) {
            nums.push_back(num);
        }

        sort(nums.begin(), nums.end());

        return (int)s.size();
    }
};
```
