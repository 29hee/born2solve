**문제 이름 또는 번호**: 2920 / 음계

**문제 링크**: https://www.acmicpc.net/problem/2920

**오늘의 메모(선택)**: 하하...9

**코드(선택)**:

```c++
#include <iostream>
#include <vector>

using namespace std;

int main() {
    vector<int> asc = {1, 2, 3, 4, 5, 6, 7, 8};
    vector<int> desc = {8, 7, 6, 5, 4, 3, 2, 1};
    
    vector<int> input;
    for (int i = 0; i < 8; i++) {
        int num;
        cin >> num;
        input.push_back(num);
    }
    
    for (int i = 0; i < 8; i++) {
        if (asc[i] != input[i]) break;
        if (i == 7) {
            cout << "ascending";
            return 0;   
        }
    }
    for (int i = 0; i < 8; i++) {
        if (desc[i] != input[i]) break;
        if (i == 7) {
            cout << "descending";
            return 0;            
        }
    }
    cout << "mixed";
}
```
