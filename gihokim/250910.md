**오늘의 메모(선택)**: 하하... 3

**코드(선택)**:

```c++
#include <iostream>

using namespace std;

int main() {
    int N, T, P;
    int tShirts[6];

    // input
    cin >> N;
    for (int i = 0; i < 6; i++) cin >> tShirts[i];
    cin >> T >> P;

    // solve
    int tAns = 0;
    for (int i = 0; i < 6; i++) {
        tAns += (tShirts[i] / T);
        if (tShirts[i] % T != 0) tAns++;
    }

    // output
    cout << tAns << '\n';
    cout << N / P << ' ' << N % P;
}
```
