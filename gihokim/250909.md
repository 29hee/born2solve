**오늘의 메모(선택)**: 하하... 2

**코드(선택)**:

```c++
#include <iostream>

using namespace std;

int main() {
    int T;
    cin >> T;
    for (int tc = 1; tc <= T; tc++) {

        int H, W, N;
        cin >> H >> W >> N;
        N--;

        int floor = N % H + 1;
        int roomNo = N / H + 1;
        
        cout << floor;
        if (roomNo < 10) cout << 0;
        cout << roomNo << '\n';
    }
}
```
