```c++
#include <cstring>

class Solution {
public:
    int climbStairs(int n) {
        int memo[46] = {0}; // ways to climb

        memo[1] = 1;
        memo[2] = 2;
        for (int i = 3; i <= 45; i++)
            memo[i] = memo[i - 2] + memo[i - 1];
        
        return memo[n];
    }
};
```
