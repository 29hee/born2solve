**코드(선택)**:
```c++
#include <iostream>

int main() {
	int N;
	int S, M, L, XL, XXL, XXXL;
	int T, P;

	std::cin >> N;
	std::cin >> S >> M >> L >> XL >> XXL >> XXXL;
	std::cin >> T >> P;

	int t_bundles = 0;
	t_bundles += (S + T - 1) / T;
	t_bundles += (M + T - 1) / T;
	t_bundles += (L + T - 1) / T;
	t_bundles += (XL + T - 1) / T;
	t_bundles += (XXL + T - 1) / T;
	t_bundles += (XXXL + T - 1) / T;
	
	int p_bundles = N / P;
	int p_singles = N % P;

	std::cout << t_bundles << std::endl;
	std::cout << p_bundles << " " << p_singles << std::endl;
	
	return 0;
}
```
