**코드(선택)**:

```cpp

#include <string>
#include <vector>
#include <algorithm>

using namespace std;

string base = "0123456789ABCDEF";

string toBaseN(int num, int n) {
    if (num == 0)
        return "0";

    string res;

    while (num > 0) {
        int remainder = num % n;
        res += base[remainder];
        num /= n;
    }

    reverse(res.begin(), res.end());
    return res;
}

string solution(int n, int t, int m, int p) {
    string answer = "";
    string res;
    
    for (int i = 0; i < t*m; i++)
        res += toBaseN(i,n);
    
    int idx=p-1;
    while (answer.size() != t) {
        answer+=res[idx];
        idx+=m;
    }
    return answer;
}

```
