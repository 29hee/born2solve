```cpp

#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> prices) {
    vector<int> answer;
    int n = prices.size();
    for (int i =0; i < n; i++) {
        int target = prices[i];
        int j = i +1;
        for (; j < n; j++) {
            if (target > prices[j]) {
                answer.push_back(j-i);
                break;
            }
        }
        if (j == n)
            answer.push_back(j-i-1);
    }
    return answer;
}
```
