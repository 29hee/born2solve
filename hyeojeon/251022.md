#include <string>
#include <vector>
#include <queue>

using namespace std;

int solution(vector<int> queue1, vector<int> queue2) {
    int answer = 0;
    queue<int> q1, q2;
    long long sum1 = 0;
    long long sum2 = 0;
    
    for (int num : queue1) {
        q1.push(num);
        sum1 += num;
    }
    for (int num : queue2) {
        q2.push(num);
        sum2 += num;
    }
    
    int max_operations = queue1.size() * 3;
    
    while (sum1 != sum2 && answer <= max_operations) {
        if (sum1 > sum2) {
            int n = q1.front();
            q1.pop();
            q2.push(n);
            sum1 -= n;
            sum2 += n;
        }
        else if (sum1 < sum2){
            int n = q2.front();
            q2.pop();
            q1.push(n);
            sum1 += n;
            sum2 -= n;
        }
        answer++;
    }
    
    if (sum1 == sum2)
        return answer;
    return -1;
}
