**오늘의 메모(선택)**: 아니 이게 합집합 교집합 구해주는 함수가 있다고

**코드(선택)**:

```cpp

#include <string>
#include <set>
#include <vector>

using namespace std;

int solution(string dirs) {
    int answer = 0;
    int x = 0;
    int y = 0;
    set<vector<int>> visit;
    
    int dx[128], dy[128];
    dx['U'] = 0; dy['U'] = 1;
    dx['D'] = 0; dy['D'] = -1;
    dx['R'] = 1; dy['R'] = 0;
    dx['L'] = -1; dy['L'] = 0;
    
    for (int i = 0; i < dirs.size(); i++) {
        int nx = x + dx[dirs[i]];
        int ny = y + dy[dirs[i]];
        
        if (nx < -5 || nx > 5 || ny < -5 || ny > 5)
            continue;
        
        int x1 = x, y1 = y, x2 = nx, y2 = ny;
        if (x1 > x2 || (x1 == x2 && y1 > y2)) {
            swap(x1, x2);
            swap(y1, y2);
        }
        
        vector<int> path = {x1, y1, x2, y2};
         if (visit.find(path) == visit.end()) {
            visit.insert(path);
            answer++;
        }
        
        x = nx;
        y = ny;
    }
    return answer;
}

```
