#include <string>
#include <vector>
#include <set>
#include <stack>

using namespace std;

int solution(int m, int n, vector<string> board) {
    int answer = 0;
    set<pair<int, int>> pos;
    
    while (1) {
        for (int i = 0; i < m -1; i++) {
            for (int j = 0; j < n -1; j++) {
                char name = board[i][j];
                if (name == ' ')
                    continue ;
                if (name == board[i+1][j] && name == board[i][j+1] && name == board[i+1][j+1]) {
                    pos.insert({i, j});
                    pos.insert({i+1, j});
                    pos.insert({i, j+1});
                    pos.insert({i+1, j+1});
                }
            }
        }
        if (pos.empty())
            break ;
        for (auto [x, y] : pos) {
            board[x][y] = ' ';
            answer++;
        }
        for (int j = 0; j < n; j++) {
            stack<char> sk;
            for (int i = 0; i < m; i++) {
                if (board[i][j] != ' ')
                    sk.push(board[i][j]);
            }
            for (int i = m-1; i >= 0; i--) {
                if (sk.empty())
                    board[i][j] = ' ';
                else {
                    board[i][j] = sk.top();
                    sk.pop();
                }
            }
        }
        pos.clear();
    }
    return answer;
}
