**오늘의 메모(선택)**: 맨해튼거리

**코드(선택)**:

```cpp

#include <string>
#include <vector>

using namespace std;

int N, M, K;
int dx[4] = {1, 0, 0, -1};
int dy[4] = {0, -1, 1, 0};
char dir[4] = {'d', 'l', 'r', 'u'};
bool find;

class Point {
    public:
        int x;
        int y;
        Point(int x, int y) {
            this->x = x;
            this->y = y;
        }
        bool operator==(const Point &op) const {
            return (op.x == x && op.y == y);
        }
};

void dfs(Point p, Point end, int step, string path, string &answer) {
    if (find)
        return;
    int remaining = K - step;
    int d = abs(p.x - end.x) + abs(p.y - end.y);

    if (d > remaining || (remaining - d) % 2 != 0)
        return;
    
    if (step == K) {
        if (p == end) {
            find = true;
            answer = path;
        }
        return ;
    }
    for (int d = 0; d < 4; ++d) {
        Point np(p.x + dx[d], p.y + dy[d]);
        if (np.x > 0 && np.x <= N && np.y > 0 && np.y <= M) {
            dfs(np, end, step + 1, path + dir[d], answer);
        }
    }
}

string solution(int n, int m, int x, int y, int r, int c, int k) {
    N = n;
    M = m;
    K = k;
    
    string answer;
    find = false;
    for (int i = 0; i < k; i++)
        answer+='z';
    Point start(x, y);
    Point end(r, c);
    
    dfs(start, end, 0, "", answer);
    if (!find)
        answer = "impossible";
    
    
    return answer;
}
```
