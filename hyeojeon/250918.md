#include <vector>
#include <algorithm>

using namespace std;

int solution(vector<vector<int> > land)
{
    int answer = 0;
    int n = land.size();
    vector<vector<int>> res(n, vector<int>(4,0));
    
    for (int i = 0; i < 4; i++) {
        res[0][i] = land[0][i];
    }

    for (int i = 1; i < n; i++) {
        for (int j = 0; j < 4; j++) {
            int val = -1;
            for (int k = 0; k < 4; k++) {
                if (k != j)
                    val = max(val, res[i-1][k]);
            }
            res[i][j] = val + land[i][j];
        }
    }
    
    answer = *max_element(res[n-1].begin(), res[n-1].end());
    
    return answer;
}
