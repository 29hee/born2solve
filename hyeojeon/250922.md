#include <string>
#include <vector>
#include <stack>

using namespace std;

int solution(vector<int> order) {
    stack<int> sub;
    int idx = 1;
    int n = order.size();
    int i = 0;
    while (idx <= n) {
        if (order[i] == idx) {
            i++;
            idx++;
        }
        else if (!sub.empty() && order[i] == sub.top()) {
            sub.pop();
            i++;
        }
        else {
            sub.push(idx);    
            idx++;
        }
    }
    while (!sub.empty() && order[i] == sub.top()) {
        sub.pop();
        i++;
    }
    return i;
}
