#include <string>
#include <vector>
#include <queue>

using namespace std;

int solution(int bridge_length, int weight, vector<int> truck_weights) {
    int answer = 0;
    int time = 0;
    int n = truck_weights.size();
    queue<int> qu;
    int totalw = 0;
    
    for (int i = 0; i < n; i++) {
        int w = truck_weights[i];
        while (1) {
            if (qu.empty()) {
                qu.push(w);
                totalw += w;
                time++;
                break;
            }
            else if (qu.size() == bridge_length) {
                totalw -= qu.front();
                qu.pop();
            }
            else {
                if (totalw + w <= weight) {
                    qu.push(w);
                    totalw += w;
                    time++;
                    break;
                } 
                else {
                    qu.push(0);
                    time++;
                }
            }
        }
        
    }
    answer = time + bridge_length;
    return answer;
}
