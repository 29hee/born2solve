#include <string>
#include <vector>
#include <algorithm>

using namespace std;

struct File {
    string head;
    int num;
    int idx;
    string name;
};

string toLower(const string& s) {
    string result = s;
    for (char& c : result) c = tolower(c);
    return result;
}

File pasrs(int idx, const string &name) {
    File file;
    file.name = name;
    file.idx = idx;
    
    int i = 0;
    int len = name.size();
    while (i < len && !isdigit(name[i]))
        i++;
    file.head = toLower(name.substr(0, i));
    
    int start = i;
    while (i < len && isdigit(name[i]) && (i - start) < 5)
        i++;
    file.num = stoi(name.substr(start, i - start));
    return file;
}

vector<string> solution(vector<string> files) {
    vector<string> answer;
    int n = files.size();
    vector<File> vec;
    
    for (int i = 0; i < n; i++) {
        vec.push_back(pasrs(i, files[i]));
    }
    
    sort(vec.begin(), vec.end(), [](const File& a, const File& b) {
        if (a.head != b.head)
            return a.head < b.head;
        if (a.num != b.num)
            return a.num < b.num;
        return a.idx < b.idx;
    });
    
    for (const auto& f : vec)
        answer.push_back(f.name);
    return answer;
}
