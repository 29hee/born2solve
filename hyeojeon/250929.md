**오늘의 메모(선택)**: 최적화가 안되네

**코드(선택)**:

```cpp

#include <string>
#include <vector>
#include <stack>
#include <list>

using namespace std;

string solution(int n, int k, vector<string> cmd) {
    string answer(n,'O');
    stack<int> st;
    list<int> lst;
    
    int len = cmd.size();
    for (int i = 0; i < n; i++)
        lst.push_back(i);
    
    auto it = lst.begin();
    advance(it, k);
    
    for (int i = 0; i < len; i++) {
        if (cmd[i][0] == 'U') {
            int x = stoi(cmd[i].substr(2));
            while (x--)
                it--;
        }
        else if (cmd[i][0] == 'D') {
            int x = stoi(cmd[i].substr(2));
            while (x--)
                it++;
        }
        else if (cmd[i][0] == 'C') {
            st.push(*it);
            it = lst.erase(it);
            if (it == lst.end())
                it--;
        }
        else if (cmd[i][0] == 'Z') {
            auto tmp = st.top();
            st.pop();
            auto insertPos = lst.begin();
            while (insertPos != lst.end() && *insertPos < tmp)
                ++insertPos;
            lst.insert(insertPos, tmp);
        }
    }
    
    while (!st.empty()) {
        auto tmp = st.top();
        st.pop();
        answer[tmp] = 'X';
    }
    return answer;
}
```
