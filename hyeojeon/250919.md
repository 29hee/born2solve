#include <string>
#include <vector>

using namespace std;

vector<vector<int>> rotate(vector<vector<int>> &mtrx)  {
    int n = mtrx.size();
    vector<vector<int>> rot(n, vector<int>(n));
    
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            rot[i][j] = mtrx[j][n-i-1];
        }
    }
    return rot;
}

bool unlock(vector<vector<int>> &temp, int n, int m) {
    bool unlock = true;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            if (temp[i + m - 1][j + m - 1] != 1) {
                return false;
            }
        }
    }
    return true;
}

bool check(vector<vector<int>> &key, vector<vector<int>> &padding, int n) {
    int m = key.size();
    int p = padding.size();
    for (int i = 0; i <= p-m; i++) {
        for (int j = 0; j <= p-m; j++) {
            for (int k = 0; k < m; k++) {
                for (int l = 0; l < m; l++) {
                    padding[i + k][j + l] += key[k][l];
                }
            }
            if (unlock(padding, n, m) == true)
                return true;
            for (int k = 0; k < m; k++) {
                for (int l = 0; l < m; l++) {
                    padding[i + k][j + l] -= key[k][l];
                }
            }
        }
    }
    return false;
}

bool solution(vector<vector<int>> key, vector<vector<int>> lock) {
    bool answer = false;
    
    int n = lock.size();
    int m = key.size();
    int p = n + 2 * (m - 1);
    vector<vector<int>> padding(p, vector<int>(p, 0));
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++)
            padding[i+m-1][j+m-1] = lock[i][j];
    }
    
    for (int i = 0; i < 4; i++) {
        if (check(key, padding, n) == true)
            return true;
        key = rotate(key);
    }
    return answer;
}
