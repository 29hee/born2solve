```cpp

#include <string>
#include <vector>
#include <algorithm>

using namespace std;

string solution(vector<int> numbers) {
    string answer = "";
    vector<string> tmp;
    for (int i : numbers)
        tmp.push_back(to_string(i));
    sort(tmp.begin(), tmp.end(), [](const string& a, const string& b) {
        return a + b > b + a;
    });
    for (string s : tmp)
        answer+=s;
    if (answer[0] == '0') {
        return "0";
    }
    return answer;
}
```
