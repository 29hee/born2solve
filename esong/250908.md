**코드(선택)**:
```cpp
#include <iostream>
#include <deque>
#include <vector>
#include <algorithm>

int	main()
{
	int	n, m;
	std::cin >> n >> m;

	std::deque<int>	numbers;
	std::vector<int>	targets(m);

	for (int i = 0; i < n; ++i)
		numbers.push_back(i + 1);
	for (int i = 0; i < m; ++i)
		std::cin >> targets[i];

	int	targetPos = 0;
	int	idx, rightDist, leftDist;
	int	moveCnt = 0;
	while (targetPos < targets.size())
	{
		if (targets[targetPos] == numbers.front())
		{
			numbers.pop_front();
			++targetPos;
		}
		else
		{
			idx = std::find(numbers.begin(), numbers.end(), targets[targetPos]) - numbers.begin();
			rightDist = idx;
			leftDist = numbers.size() - idx;

			if (rightDist < leftDist)
			{
				numbers.push_back(numbers.front());
				numbers.pop_front();
				++moveCnt;
			}
			else
			{
				numbers.push_front(numbers.back());
				numbers.pop_back();
				++moveCnt;
			}
		}
	}

	std::cout << moveCnt;

	return 0;
}
```
