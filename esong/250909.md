**코드(선택)**:
```cpp

#include <iostream>

int		check(std::string &str, int left, int right);
bool	isPalindrome(std::string &str, int left, int right);

int	main()
{
	int	n, result;
	int left, right;
	std::string	str;

	std::cin >> n;
	while (n--)
	{
		result = 0;
		std::cin >> str;
		left = 0;
		right = str.size() - 1;
		while (left < right)
		{
			if (str[left] == str[right])
			{
				++left;
				--right;
			}
			else
			{
				result = check(str, left, right);
				break ;
			}
		}
		std::cout << result << "\n";
	}
}

int	check(std::string &str, int left, int right)
{
	if (left + 1 <= right && str[left + 1] == str[right])
	{
		if (isPalindrome(str, left + 1, right) == true)
			return 1;
	}

	if (left <= right - 1 && str[left] == str[right - 1])
	{
		if (isPalindrome(str, left, right - 1) == true)
			return 1;
	}

	return 2;
}

bool	isPalindrome(std::string &str, int left, int right)
{
	while (left < right)
	{
		if (str[left] == str[right])
		{
			++left;
			--right;
		}
		else
		{
			return false;
		}
	}
	return true;
}
```
