#include <iostream>
#include <cstring>
using namespace std;

int T, N, M;
int result;
bool relation[21][21];
bool visited[21];

void init() {
    result = 0;
    memset(relation, false, sizeof(relation));
    memset(visited, false, sizeof(visited));
}

bool canSelect(int idx) {
    for (int i = 1; i < idx; i++) {
        if (visited[i] && relation[i][idx])
            return false;
    }
    return true;
}

void dfs(int idx) {
    result++;

    for (int i = idx; i <= N; i++) {
        if (!visited[i] && canSelect(i)) {
            visited[i] = true;
            dfs(i + 1);
            visited[i] = false;
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> T;
    for (int tc = 1; tc <= T; tc++) {
        cin >> N >> M;
        init();

        int a, b;
        for (int i = 0; i < M; i++) {
            cin >> a >> b;
            relation[a][b] = true;
            relation[b][a] = true;
        }

        dfs(1);
        cout << "#" << tc << " " << result << "\n";
    }

    return 0;
}
