오늘의 메모(선택): 냅색 알고리즘을 적용하여 풀기

코드(선택): cpp

```cpp
#include <iostream>
#include <algorithm>
using namespace std;

int *score, *cal, **dp;
int n, maxCal;

void init() {
	score = new int[n + 1];
	cal = new int[n + 1];
	dp = new int *[n + 1];
	for (int i = 0; i <= n; i++) {
		dp[i] = new int[maxCal + 1]();
	}
}

void destroy() {
	delete[] score;
	delete[] cal;
	for (int i = 0; i <= n; i++) delete[] dp[i];
	delete[] dp;
}

int main() {
	int t;

	cin >> t;
	for (int tc = 1; tc <= t; tc++) {
		cin >> n >> maxCal;
		init();
		for (int i = 0; i < n; i++) {
			cin >> score[i + 1] >> cal[i + 1];
		}
		for (int i = 1; i <= n; i++) {
			for (int j = 1; j <= maxCal; j++) {
				if (j >= cal[i]) {
					dp[i][j] = max(dp[i - 1][j], dp[i - 1][j - cal[i]] + score[i]);
				}
				else {
					dp[i][j] = dp[i - 1][j];
				}
			}
		}
		cout << "#" << tc << " " << dp[n][maxCal] << "\n";
		destroy();
	}
}
```
