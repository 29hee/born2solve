**오늘의 메모(선택)**: 
                순서대로 진행하길래 queue 씀

**코드(선택)**:

C++

```

#include <string>
#include <vector>
#include <iostream>
#include <queue>
#include <unordered_set>

using namespace std;

bool isInvalidWord(unordered_set<string>& usedWord, string& prevWord, string& currentWord);

vector<int> solution(int n, vector<string> words) {
    vector<int> answer;
    queue<int> order;
    
    for (int i = 1; i <= n; ++i)
    {
        order.push(i);
    }
    
    int gameRound = 1;
    int remainTurn = order.size();
    unordered_set<string> usedWord;
    string prevWord = "";
    for (int i = 0; i < words.size(); ++i)
    {
        int turn = order.front();
        order.pop();
        --remainTurn;
        
        string& word = words[i];
        if (isInvalidWord(usedWord, prevWord, word))
        {
            answer.push_back(turn);
            answer.push_back(gameRound);
            return answer;
        }
        prevWord = word;
        order.push(turn);
        usedWord.insert(word);
        if (remainTurn == 0)
        {
            ++gameRound;
            remainTurn = order.size();
        }
    }
    
    answer.push_back(0);
    answer.push_back(0);
    return answer;
}

bool isInvalidWord(unordered_set<string>& usedWord, string& prevWord, string& currentWord)
{
    return currentWord.length() <= 1 
        || usedWord.find(currentWord) != usedWord.end() 
        || prevWord != "" && prevWord[prevWord.size() - 1] != currentWord[0];
}
```
