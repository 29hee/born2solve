                    단순 구현 문제

**코드(선택)**:

C++

```

#include <string>
#include <vector>
#include <unordered_map>
#include <algorithm>

using namespace std;

int solution(int k, vector<int> tangerine) {
    
    unordered_map<int, int> map;
    
    for (size_t i = 0; i < tangerine.size(); ++i)
    {
        ++map[tangerine[i]];
    }

    vector<int> bucket;
    for (auto pair : map)
    {
        bucket.push_back(pair.second);
    }
    sort(bucket.begin(), bucket.end());
    
    int answer = 0;
    int remainSize = k;
    for (int i = bucket.size() - 1; i >= 0; --i)
    {
        ++answer;
        remainSize -= bucket[i];
        if (remainSize <= 0)
        {
            break;
        }
    }
    
    return answer;
}
```
