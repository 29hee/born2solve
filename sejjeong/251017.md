#include <string>
#include <vector>
#include <unordered_map>

using namespace std;

bool CanBuy(unordered_map<string, int>& itemCheckList);

int solution(vector<string> want, vector<int> number, vector<string> discount) {
    // 일정한 금액 지불 -> 회원 자격
    // 회원에게 매일 한 가지 제품 할인, 하루에 하나 씩만 구매 가능
    // 자기가 원하는 제품과 수량이 10일 안으로 살 수 있는 경우를 맞춘다.
    
    unordered_map<string, int> wantItem;
    size_t itemSize = 0;
    for (size_t i = 0; i < want.size(); ++i)
    {
        wantItem.insert(pair<string, int>(want[i], number[i]));
        itemSize += static_cast<size_t>(number[i]);
    }
    
    // 물품의 길이는 10개 이하
    // 많이 연산해도 100만
    // 순서가 중요
    
    int answer = 0;
    for (size_t i = 0; i + itemSize - 1 < discount.size(); ++i)
    {
        unordered_map<string, int> itemCheckList = wantItem;
        for (size_t j = 0; j < itemSize; ++j)
        {
            unordered_map<string, int>::iterator it = itemCheckList.find(discount[i + j]);
            if (it == itemCheckList.end())
            {
                continue;
            }
            pair<const string, int>& item = *it;
            --item.second;
        }
        
        if (CanBuy(itemCheckList) == false)
        {
            continue;
        }
        
        ++answer;
    } 
    return answer;
}

bool CanBuy(unordered_map<string, int>& itemCheckList)
{
    for (auto item : itemCheckList)
    {
        if (item.second > 0)
        {
            return false;
        }
    }
    
    return true;
}
