```
#include <string>
#include <vector>
#include <unordered_set>
#include <iostream>

using namespace std;

int solution(int N, int number) {
    
    if (N == number)
    {
        return 1;
    }
    
    vector<unordered_set<int>> lookupTable;
    unordered_set<int> s;
    s.insert(N);
    lookupTable.push_back(s);
    
    const int MAX_COUNT = 8;
    const int MAX_NUMBER = 32000;
    
    for (int i = 2; i <= MAX_COUNT; ++i)
    {
        unordered_set<int> set;
        
        int ns = N;
        for (int j = 0; j < i - 1; ++j)
        {
            ns = ns * 10 + N;
        }
        
        if (ns == number)
        {
            return i;
        }

        if (ns <= MAX_NUMBER)
        {
            set.insert(ns);
        }
        
        int up = i - 2;
        int down = 0;
        while (up >= 0)
        {
            for (int upInt : lookupTable[up])
            {
                for (int downInt : lookupTable[down])
                {
                    int calculatedNum1 = upInt + downInt;
                    int calculatedNum2 = upInt * downInt;
                    int calculatedNum3 = upInt - downInt;
                    int calculatedNum4 = 0;
                    if (downInt != 0)
                    {
                        calculatedNum4 = upInt / downInt;
                    }
                               
                    
                    if (calculatedNum1 == number 
                        || calculatedNum2 == number 
                        || calculatedNum3 == number 
                        || calculatedNum4 == number)
                    {
                        return i;
                    }
                    if (calculatedNum1 <= MAX_NUMBER)
                    {
                        set.insert(calculatedNum1);
                    }
                    if (calculatedNum2 <= MAX_NUMBER)
                    {
                        set.insert(calculatedNum2);
                    }
                    if (calculatedNum3 <= MAX_NUMBER)
                    {
                        set.insert(calculatedNum3);
                    }
                    if (calculatedNum4 <= MAX_NUMBER)
                    {
                        set.insert(calculatedNum4);
                    }
                    
            
                }
            }
            --up;
            ++down;
        }
        lookupTable.push_back(set);
    }
    
    
    // 1개의 N으로 표현 가능한 number
        // N
    
    // 2개의 N으로 표현 가능한 number
        // N * N, N + N, N / N, NN, N - N
    
    // 2개의 N으로 표현 가능한 number를 가지고
    // 3개의 N으로 표현 가능한 number 구하기
    
    // 2개의 N으로 표현 가능한 number와  2 2 
    // 3개의 N으로 표현 가능한 number를 가지고  3 1
    // 4개의 N으로 표현 가능한 number 구하기
    
    
    // 4 1, 3 2, 2 3
    // 5 1, 3 3, 4 2, 2 4
    // 6 1, 3 4, 4 3, 5 2, 2 5, 1 6 
    // 7 1, 4 4, 3 5, 5 3, 2 6, 6 2, 1 7
    
    // 1, 11, 111 or N NN NNN, (NN + N)  NN + NN 66 
    
    
    return -1;
}
```
